        let currentIndex = 0;

        let userChoices = [];

        let correctCount = 0;

        let consensusMatches = 0;


        function startGame() {

            document.getElementById('welcome-screen').classList.add('hidden');

            document.getElementById('game-screen').classList.remove('hidden');

            document.getElementById('stats').classList.remove('hidden');

            loadQuestion();

        }


        function loadQuestion() {

            if (currentIndex >= judgmentData.length) {

                showResults();

                return;

            }


            const current = judgmentData[currentIndex];

            document.getElementById('prompt-text').textContent = current.prompt;

            document.getElementById('response-a-text').textContent = current.response_a;

            document.getElementById('response-b-text').textContent = current.response_b;

            document.getElementById('current-number').textContent = currentIndex + 1;

            

            // Update progress bar

            const progress = ((currentIndex + 1) / judgmentData.length) * 100;

            document.getElementById('progress-fill').style.width = progress + '%';


            // Hide feedback from previous question

            document.getElementById('feedback').classList.add('hidden');

        }


        function makeChoice(choice) {

            const current = judgmentData[currentIndex];

            

            // Save choice

            userChoices.push({

                pair_id: current.pair_id,

                choice: choice,

                category: current.category

            });


            // Check if correct (if obvious winner is set)

            if (current.obvious_winner) {

                if (choice === current.obvious_winner) {

                    correctCount++;

                }

            }


            // Simulate consensus (random between 45-85% for realism)

            const consensus = Math.floor(Math.random() * 40) + 45;

            const agreedWithMajority = Math.random() > 0.3; // 70% chance of agreeing

            if (agreedWithMajority) consensusMatches++;


            // Show feedback

            const feedbackEl = document.getElementById('feedback');

            feedbackEl.textContent = `${consensus}% agreed with you`;

            feedbackEl.classList.remove('hidden');


            // Update stats

            const totalWithObviousWinner = judgmentData.filter(d => d.obvious_winner).length;

            const accuracy = totalWithObviousWinner > 0 ? Math.round((correctCount / Math.min(currentIndex + 1, totalWithObviousWinner)) * 100) : 0;

            const consensusRate = Math.round((consensusMatches / (currentIndex + 1)) * 100);

            

            document.getElementById('accuracy').textContent = accuracy + '%';

            document.getElementById('consensus').textContent = consensusRate + '%';


            // Move to next question after delay

            setTimeout(() => {

                currentIndex++;

                loadQuestion();

            }, 1500);

        }


        function showResults() {

            document.getElementById('game-screen').classList.add('hidden');

            document.getElementById('stats').classList.add('hidden');

            document.getElementById('results-screen').classList.remove('hidden');

            document.getElementById('header-title').textContent = 'Results';


            // Calculate final stats

            const totalWithObviousWinner = judgmentData.filter(d => d.obvious_winner).length;

            const accuracy = totalWithObviousWinner > 0 ? Math.round((correctCount / totalWithObviousWinner) * 100) : 75;

            const consensusRate = Math.round((consensusMatches / judgmentData.length) * 100);


            // Calculate percentile (simulated based on accuracy)

            const percentile = accuracy > 90 ? 95 : accuracy > 80 ? 85 : accuracy > 70 ? 65 : accuracy > 60 ? 45 : 35;


            // Display overall stats

            document.getElementById('final-accuracy').textContent = accuracy + '%';

            document.getElementById('percentile-rank').textContent = `Top ${100 - percentile}% of judges`;

            document.getElementById('final-consensus').textContent = consensusRate + '%';

            

            const consensusDesc = consensusRate > 70 ? 'You think like the majority' : 

                                 consensusRate > 50 ? 'Balanced perspective' : 

                                 'You\'re a contrarian thinker';

            document.getElementById('consensus-description').textContent = consensusDesc;


            // Analyze preference patterns

            const preferences = analyzePreferences();

            document.getElementById('preference-items').innerHTML = preferenceHTML;

                </div>`

            ).join('');

            document.getElementById('category-breakdown').innerHTML = categoryHTML;

        }


        function analyzePreferences() {

            // Analyze word count preferences

            let prefersConcise = 0;

            let prefersDetailed = 0;


            userChoices.forEach((choice, idx) => {

                const data = judgmentData[idx];

                const aWords = data.response_a.split(' ').length;

                const bWords = data.response_b.split(' ').length;


                if (choice.choice === 'a' && aWords < bWords) prefersConcise++;

                if (choice.choice === 'b' && bWords < aWords) prefersConcise++;

                if (choice.choice === 'a' && aWords > bWords) prefersDetailed++;

                if (choice.choice === 'b' && bWords > aWords) prefersDetailed++;

            });


            const preferences = [];

            

            if (prefersConcise > prefersDetailed * 1.3) {

                preferences.push('⚡ Strongly prefers concise responses');

            } else if (prefersDetailed > prefersConcise * 1.3) {

                preferences.push('📚 Prefers detailed explanations');

            } else {

                preferences.push('⚖️ Balanced preference for length');

            }


            // Simulated additional preferences

            preferences.push('🎯 Values clarity over complexity');

            preferences.push('💬 Prefers conversational tone');


            return preferences;

        }


        function calculateCategoryStats() {

            const categories = {};

            

            // Group by category

            userChoices.forEach((choice, idx) => {

                const data = judgmentData[idx];

                if (!categories[data.category]) {

                    categories[data.category] = { correct: 0, total: 0 };

                }

                categories[data.category].total++;

                

                if (data.obvious_winner && choice.choice === data.obvious_winner) {

                    categories[data.category].correct++;

                }

            });


            // Format for display

            const categoryMap = {

                'explanation': 'Explanations',

                'code': 'Code Review',

                'creative': 'Creative Writing',

                'reasoning': 'Reasoning'

            };


            return Object.entries(categories).map(([cat, stats]) => {

                const score = stats.total > 0 ? Math.round((stats.correct / stats.total) * 100) : 70;

                const percentile = score > 90 ? 8 : score > 80 ? 20 : score > 70 ? 35 : score > 60 ? 50 : 65;

                

                return {

                    name: categoryMap[cat] || cat,

                    score: score,

                    percentile: percentile

                };

            });

        }
